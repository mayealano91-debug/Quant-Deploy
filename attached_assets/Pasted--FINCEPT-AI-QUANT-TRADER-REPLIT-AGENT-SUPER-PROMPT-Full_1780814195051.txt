// 3. CATALYST: Check for unusual volume (>2x 20-period average)

### CORS Solution:
```javascript
// Alpaca API blocks direct browser requests for some endpoints
// Create a lightweight backend proxy:
// POST /api/proxy → forwards to Alpaca with stored credentials
// This keeps API keys server-side and solves CORS

// Express proxy example:
app.post('/api/alpaca/*', async (req, res) => {
  const alpacaUrl = `${process.env.ALPACA_BASE_URL}${req.path.replace('/api/alpaca', '')}`;
  const response = await fetch(alpacaUrl, {
    method: req.method,
    headers: {
      'APCA-API-KEY-ID': process.env.ALPACA_API_KEY,
      'APCA-API-SECRET-KEY': process.env.ALPACA_SECRET_KEY,
      'Content-Type': 'application/json'
    },
    body: req.method !== 'GET' ? JSON.stringify(req.body) : undefined
  });
  res.json(await response.json());
});
```
